\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving the graph of a polynomial function.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of graph of a polynomial function? 
%2
\item How can you apply your knowledge of graph of a polynomial function in solving these real life problems? 

\end{enumerate}   
}

\end{enumerate}
