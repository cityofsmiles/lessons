	\item %\hspce The first term of a geometric sequence is -2, and the third term is $-\frac{1}{2}$. Find the fifth term. 
$a_1=-2$, $\displaystyle a_3=-\frac{1}{2}$,\\
$a_5=?$, $r=? $, $n=5$\\
$r=\sqrt[n-k]{\displaystyle \frac{a_n}{a_k} }$\\
$r=\sqrt[3-1]{\displaystyle \frac{a_3}{a_1} }$\\
$r=\sqrt[2]{\displaystyle \frac{-\frac{1}{2}}{-2} }$\\
$r=\sqrt{\displaystyle \frac{1}{4} }$\\
$r=\displaystyle \frac{1}{2}$\\
$a_n=a_1r^{n-1}$\\	 
$a_5=(-2)\left(\displaystyle \frac{1}{2}\right)^{5-1}$\\	 
$a_5=(-2)\left(\displaystyle \frac{1}{2}\right)^{4}$\\	 
$a_5=(-2)\left(\displaystyle \frac{1}{16}\right)$\\	
$a_5=-\displaystyle \frac{1}{8}$\\	 

	\item %\hspce The common ratio in a geometric sequence is $\frac{2}{3}$ and the fourth term is 1. Find the third term.
$a_4=1$, $a_3=? $, $n=3$\\
$a_1=?$, $r=\displaystyle \frac{2}{3}$\\ 
$a_n=a_1r^{n-1}$\\	 
$a_4=a_1\left(\displaystyle \frac{2}{3}\right)^{4-1}$\\	 
$1=a_1\left(\displaystyle \frac{2}{3}\right)^{3}$\\	 
$1=a_1\left(\displaystyle \frac{8}{27}\right) $\\	
$a_1= \displaystyle \frac{27}{8}$\\
$a_n=a_1r^{n-1}$\\	 
$a_3=\left( \displaystyle \frac{27}{8}\right)\left( \displaystyle \frac{2}{3}\right)^{3-1}$\\	 
$a_3=\left( \displaystyle \frac{27}{8}\right)\left( \displaystyle \frac{2}{3}\right)^{2}$\\	
$a_3=\left( \displaystyle \frac{27}{8}\right)\left( \displaystyle \frac{4}{9}\right)$\\	
$a_3=\left( \displaystyle \frac{108}{72}\right)$  

	\item %\hspce The common ratio in a geometric sequence is $\frac{3}{4}$ and the fifth term is $\frac{81}{16}$. Find the first three terms.
$a_1=? $, $a_2=?$,$a_3=?$,\\
$ a_5=\displaystyle\frac{81}{16}$,
$r=\displaystyle \frac{3}{4}$\\
$a_n=a_1r^{n-1}$\\	 
$a_5=a_1\left( \displaystyle \frac{3}{4}\right) ^{5-1}$\\	 
$\displaystyle\frac{81}{16}=a_1\left( \displaystyle \frac{3}{4}\right)^{4}$\\	 
$\displaystyle\frac{81}{16}=a_1\left( \displaystyle \frac{81}{256}\right)$\\
$a_1=16$\\
$a_2=(16)\left( \displaystyle \frac{3}{4}\right)=12 $\\
$a_3=(12)\left( \displaystyle \frac{3}{4}\right)=9 $

	\item %\hspce Which term of the geometric sequence 3, 6, 12,... is 768?
$a_1=3$, $a_n=768$,\\
$r=6\div 3=2$, $n=? $\\
$a_n=a_1r^{n-1}$\\	 
$\displaystyle \frac{768}{3} =\displaystyle \frac{(3)(2)^{n-1}}{3} $\\	 
$256=2^{n-1}$\\
$2^{8}=2^{n-1}$\\
$8=n-1$\\
$n=9$
 
	\item %\hspce The common ratio in a geometric sequence is $\frac{3}{2}$ and the fifth term is 1. Find the first three terms.
$a_1=? $, $a_2=? $, $a_3=? $,\\
$a_5=1$, $r=\displaystyle \frac{3}{2}$\\
$a_n=a_1r^{n-1}$\\	
$a_5=a_1\left(\displaystyle \frac{3}{2} \right)^{5-1}$\\	
$1=a_1\left(\displaystyle \frac{3}{2} \right)^{4}$\\	
$1=a_1\left(\displaystyle \frac{81}{16} \right)$\\	
$a_1=\displaystyle \frac{16}{81}$\\
$a_2=\left( \displaystyle \frac{16}{81}\right)\left( \displaystyle \frac{3}{2}\right) =\displaystyle \frac{8}{27} $\\

$a_3=\left( \displaystyle \frac{8}{27}\right)\left( \displaystyle \frac{3}{2}\right) =\displaystyle \frac{4}{9}   $
