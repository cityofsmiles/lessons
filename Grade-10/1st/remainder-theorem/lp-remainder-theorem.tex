\def \ObjectivesDay {
\item Demonstrate  the remainder theorem;

\item Exhibit determination and self-reliance in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving the remainder theorem.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of the remainder theorem? 
%2
\item How can you apply your knowledge of remainder theorem in solving these real life problems? 

\end{enumerate}   
}

\begin{enumerate}[label = \arabic*. ]

\end{enumerate}
