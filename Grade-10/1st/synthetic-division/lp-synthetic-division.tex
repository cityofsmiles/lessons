\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving synthetic division.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of synthetic division?

\item How can you apply your knowledge of synthetic division in solving these real life problems? 

\end{enumerate}   
}
