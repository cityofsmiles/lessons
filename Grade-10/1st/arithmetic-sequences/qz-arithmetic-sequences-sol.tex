\begin{enumerate}
\begin{multicols}{2}

\item %The $101^{th}$ term of the arithmetic sequence if 
$a_{1}=-5$, $d=-4$,\\
$n=101$,  $a_{101}=?$\\
$a_n=a_1+(n-1)d$\\
$a_{101}=-5+(101-1)(-4)$\\
$a_{101}=-405$

\item %The $39^{th}$ term of the arithmetic sequence if $a_{1}$ = 40 and $d$ = $\frac{1}{2}$.
$a_{1}=40$, $d=\displaystyle \frac{1}{2} $,\\
$n=39$,  $a_{39}=?$\\
$a_n=a_1+(n-1)d$\\
$a_{39}=40+(39-1)\left(\displaystyle \frac{1}{2}\right)$\\
$a_{39}=59$


\item  %In the sequence 6, 10, 14,..., what term has a value of 286?
$a_{1}=6$, $d=10-6=4 $,\\
$n=? $,  $a_{n}=286$\\
$a_n=a_1+(n-1)d$\\
$286=6+(n-1)(4)$\\
$286=6+4n-4$\\
$286=4n+2$\\
$286-2=4n$\\
$\displaystyle \frac{284}{4} =\displaystyle \frac{4n}{4} $\\
$n=71$


\item %The $1^{st}$ term of the sequence if  $a_{5}$ = 26 and $a_{12}$ = 47.
$a_{5} = 26$, $a_{12} = 47$,\\
$d=? $, $a_1=? $\\
$d = \displaystyle \frac{a_{n} - a_{k}}{n - k} $\\
$d = \displaystyle \frac{a_{12} - a_{5}}{12 - 5} $\\
$d = \displaystyle \frac{47 - 26}{7} $\\
$d = \displaystyle \frac{21}{7} $\\
$d=3$\\
$a_n=a_1+(n-1)d$\\
$a_5=a_1+(5-1)(3)$\\
$26=a_1+12$\\
$26-12=a_1$\\
$a_1=14$\\

\item %The $61^{th}$ term of the sequence if  $a_{12}$ = 8 and $a_{21}$ = 26.
$a_{12} = 8$, $a_{21} = 26$,\\
$n=61$, $d=? $, $a_{61}=? $\\
$d = \displaystyle \frac{a_{n} - a_{k}}{n - k} $\\
$d = \displaystyle \frac{a_{21} - a_{12}}{21 - 12} $\\
$d = \displaystyle \frac{26 - 8}{9} $\\
$d = \displaystyle \frac{18}{9} $\\
$d=2$\\
$a_n=a_1+(n-1)d$\\
$a_{12}=a_1+(12-1)(2)$\\
$8=a_1+22$\\
$8-22=a_1$\\
$a_1=-14$\\
$a_n=a_1+(n-1)d$\\
$a_{61}=-14+(61-1)(2)$\\
$a_{61}=106$

\end{multicols} 
\end{enumerate}
