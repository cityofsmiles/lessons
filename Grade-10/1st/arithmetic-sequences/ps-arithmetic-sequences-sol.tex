A.   
\begin{enumerate}
\begin{multicols}{2}

\item \hspce  $3, 5, 7,... a_{21}$\\
$a_1=3$, $n=21$,\\
$d=5-3=2$\\
$a_{21}=3+(21-1)(2)$\\
$a_{21}=43$

\item \hspce $1.4, 4.5, 7.6,... a_{51}$\\
$a_1=1.4$, $n=51$,\\
$d=4.5-1.4=3.1$\\
$a_n=a_1+(n-1)d$\\
$a_{51}=1.4+(51-1)(3.1)$\\
$a_{51}=156.4$

\item \hspce $x-2, 4x, 7x+2,... a_{12}$\\
$a_1=x-2$, $n=12$,\\
$d=4x-(x-2)$\\
$d=3x+2$\\
$a_n=a_1+(n-1)d$\\
$a_{12}=x-2+(12-1)(3x+2)$\\
$a_{12}=34x+20$

\item \hspce $14, 6, -2,... a_{28}$\\
$a_1=14$, $n=28$,\\
$d=6-14=-8$\\
$a_n=a_1+(n-1)d$\\
$a_{28}=14+(28-1)(-8)$\\
$a_{28}=-202$

\item \hspce $5, -1, -7,... a_{18}$\\
$a_1=5$, $n=18$, \\
$d=-1-5=-6$\\
$a_n=a_1+(n-1)d$\\
$a_{18}=5+(18-1)(-6)$\\
$a_{18}=-97$

\end{multicols} 
\end{enumerate}

B.  
\begin{enumerate}
\begin{multicols}{2}

\item  \hspce $a_{8} = 5$, $a_{21} = -60$,\\
$n=17$, $a_{17}=? $\\
$d = \displaystyle \frac{a_{n} - a_{k}}{n - k} $\\
$d = \displaystyle \frac{a_{21} - a_{8}}{21 - 8} $\\
$d = \displaystyle \frac{ -60 - 5}{13} $\\
$d=-5$\\
$a_n=a_1+(n-1)d$\\
$a_{8}=a_1+(8-1)(-5)$\\
$5=a_1-35$\\
$a_1=40$\\
$a_n=a_1+(n-1)d$\\
$a_{17}=40+(17-1)(-5) $\\
$a_{17}=-40$

\item \hspce $a_{15} = 29$, $a_{27} = 47$,\\
$n=5$, $a_{5}=? $\\
$d = \displaystyle \frac{a_{n} - a_{k}}{n - k} $\\
$d = \displaystyle \frac{a_{27} - a_{15}}{27 - 15} $\\
$d = \displaystyle \frac{47-29}{12} $\\
$d=1.5$\\
$a_n=a_1+(n-1)d$\\
$a_{15}=a_1+(15-1)(1.5)$\\
$29=a_1+21$\\
$a_1=8$\\
$a_n=a_1+(n-1)d$\\
$a_{5}=8+(5-1)(1.5) $\\
$a_{5}=14$

\item \hspce $a_{24} = 85$, $a_{28} = 100$,\\
$n=1$, $a_{1}=? $\\
$d = \displaystyle \frac{a_{n} - a_{k}}{n - k} $\\
$d = \displaystyle \frac{a_{28} - a_{24}}{28 - 24} $\\
$d = \displaystyle \frac{100-85}{4} $\\
$d=3.75$\\
$a_n=a_1+(n-1)d$\\
$a_{24}=a_1+(24-1)(3.75)$\\
$85=a_1+86.25$\\
$a_1=-1.25$

\item \hspce $a_{1} = -4$, $a_{25} = -100$,\\
$n=101$, $a_{101}=? $\\
$d = \displaystyle \frac{a_{n} - a_{k}}{n - k} $\\
$d = \displaystyle \frac{a_{25} - a_{1}}{25 - 1} $\\
$d = \displaystyle \frac{-100-(-4)}{24} $\\
$d=-4$\\
$a_n=a_1+(n-1)d$\\
$a_{101}=-4+(101-1)(-4)$\\
$a_{101}=-404$

\end{multicols} 
\end{enumerate}
