\item \phantom{b} %Find the specified term of each harmonic sequence. 
\begin{enumerate}[label = \arabic*. ]
%begin{multicols}{2}

%#1
\item %\hspce $\displaystyle \frac{1}{2}, \frac{1}{4}, \frac{1}{6},... a_7$ 
$2, 4, 6,\ldots  a_7$ \\
$d=4-2=2$\\
$a_{n}=a_1+(n-1)d$\\
$a_{7}=2+(7-1)(2) $\\
$a_{7}=2+(6)(2) $\\
$a_{7}=14 $\\
$a_{7}=\dfrac{1}{14} $

%#2
\item %\hspce $\displaystyle \frac{1}{5}, \frac{1}{10}, \frac{1}{15},... a_{10}$ 
$5, 10, 15, \ldots a_{10}$ \\
$d=10-5=5$\\
$a_{n}=a_1+(n-1)d$\\
$a_{10}=5+(10-1)(5) $\\
$a_{10}=5+(9)(5) $\\
$a_{10}=50 $\\
$a_{10}=\dfrac{1}{50}  $

%#3
\item %\hspce $\displaystyle \frac{1}{4}, \frac{1}{11}, \frac{1}{18},... a_9$
$4, 11, 18,\ldots  a_9$\\
$d=11-4=7$\\
$a_{n}=a_1+(n-1)d$\\
$a_{9}=4+(9-1)(7) $\\
$a_{9}=4+(8)(7) $\\
$a_{9}=60 $\\
$a_{9}=\dfrac{1}{60}  $

%#4
\item %\hspce $\displaystyle -\frac{1}{10}, -\frac{1}{3}, \frac{1}{4},... a_{14}$

$a_{14}=-10+(14-1)(7) $\\
$a_{14}=-10+(13)(7) $\\

$a_{14}=\displaystyle \frac{1}{81}  $

%#5
