
\def \LessonDay {Harmonic Sequence}

\def \ObjectivesDay {
\item Describe the similarities between harmonic sequences and arithmetic sequences; 
\item Calculate the next terms  of a given harmonic sequence; and, 
\item Project independence and interest in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving harmonic sequence.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of harmonic sequence? 
%2
\item How can you apply your knowledge of harmonic sequence in solving these real life problems? 

\end{enumerate}   
}

\begin{enumerate}[label = \arabic*. ]

\item In your own words, what is a harmonic sequence? 
%2

\end{enumerate}
