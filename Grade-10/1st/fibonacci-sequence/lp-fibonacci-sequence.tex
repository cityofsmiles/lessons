
\def \LessonDay {Fibonacci Sequence}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Describe the steps in finding the next terms of a  fibonacci sequence; 
\item Solve the next terms of a  fibonacci sequence; and, 
\item Show willingness and interest in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving fibonacci sequence.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of fibonacci sequence? 
%2
\item How can you apply your knowledge of fibonacci sequence in solving these real life problems? 

\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, what is a fibonacci sequence? 
%2
\item How do we solve problems involving fibonacci sequence? 

\end{enumerate}   
}
