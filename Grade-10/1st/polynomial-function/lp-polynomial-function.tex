
\def \LessonDay {Polynomial Function}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Describe a polynomial function; 
\item Determine the kind, the degree, the leading coefficient, and the constant term of a  polynomial function; and, 
\item Exhibit  patience and self-reliance  in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving polynomial functions.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of polynomial functions? 
%2
\item How can you apply your knowledge of polynomial functions in solving these real life problems? 

\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, what is a polynomial function? 
%2
\item How do we solve problems involving polynomial functions? 

\end{enumerate}   
}
