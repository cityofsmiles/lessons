\item \hspce $a_{n} = 2n - 1$
\item \hspce $a_{n} = 12 - 3n$
\item \hspce $a_{n} = 3^{n}$
\item \hspce $a_{n} = -2^{n}$

\end{multicols} 
\end{enumerate}
