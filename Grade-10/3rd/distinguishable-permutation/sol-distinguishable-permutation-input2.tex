\item %In how many ways can 4 green marbles and 6 blue marbles be arranged in a row?
$n= 10$ \redcheck \\
$p=4 $ \redcheck \\
$q= 6$ \redcheck \\
$P= \dfrac{n!}{p!q!} $ \redcheck \\
$P= \dfrac{10!}{4!6!} $ \redcheck \\
$= \dfrac{(10)(9)(8)(7)(6!)}{(4)(3)(2)(1)(6!)} $ \redcheck \\
=\fbox{$210 $} \redcheck 
ways \redcheck 
 
%2
\item %How many distinguishable permutations are possible with all the letters of the word ELLIPSES?
$n=8 $ \redcheck \\
$p= 2$ \redcheck \\
$q= 2$ \redcheck \\
$r=2 $ \redcheck \\
$P= \dfrac{n!}{p!q!r!} $ \redcheck \\
$P= \dfrac{8!}{2!2!2!} $ \redcheck \\
$= \dfrac{(8)(7)(6)(5)(4)(3)(2!)}{(2!)(2)(1)(2)(1)} $ \redcheck \\
=\fbox{$ 5,040$} \redcheck 
distinguishable permutations \redcheck 
 
%3
\item %Find the number of distinguishable permutations of the digits of the number 348,838.
$n= 6$ \redcheck \\
$p= 2$ \redcheck \\
$q= 3$ \redcheck \\
$P= \dfrac{n!}{p!q!} $ \redcheck \\
$P= \dfrac{6!}{2!3!} $ \redcheck \\
$= \dfrac{(6)(5)(4)(3!)}{(2)(1)(3!)} $ \redcheck \\
=\fbox{$60 $} \redcheck 
distinguishable permutations \redcheck 
 
\end{enumerate}
