\item %Anabelle would like to invite 9 friends to go on a trip but has room for only 5 of them. In how many ways can they be chosen?
$n= 9$ \redcheck \\
$r= 5$ \redcheck \\
$ C(n, r) = \dfrac{n!}{(n-r)!r!} $ \redcheck \\
$ C(9, 5) = \dfrac{9!}{(9-5)!5!} $ \redcheck \\
$ = \dfrac{9!}{4!5!} $ \redcheck \\
%$ = \dfrac{(9)(8)(7)(6)(5!)}{(4)(3)(2)(1)(5!)} $ \redcheck \\
=\fbox{$ 126$} \redcheck 
ways \redcheck 

%2
\item %A box with 12 articles contains one that is defective. In how many ways can Martin select 5 articles such that: 
\begin{enumerate}[label = \alph*. ]
%A
\item %the defective article is included?
$n_1= 11$ \redcheck \\
$r_1= 4$ \redcheck \\
$n_2= 1$ \redcheck \\
$r_2= 1$ \redcheck \\
$ C(11, 4) \cdot C(1, 1) = \dfrac{11!}{(11-4)!4!} -  \dfrac{1!}{(1-1)!1!} $ \redcheck \\
$ = \dfrac{11!}{7!4!} \cdot  \dfrac{1!}{0!1!} $ \redcheck \\

=\fbox{$ 330$} \redcheck 
ways \redcheck 

%B
\item %the defective article is not included?
$n_1= 11$ \redcheck \\
$r_1= 5$ \redcheck \\
$n_2= 1$ \redcheck \\
$r_2= 0$ \redcheck \\
$ C(11, 5) \cdot C(1, 0) = \dfrac{11!}{(11-5)!5!} -  \dfrac{1!}{(1-0)!0!} $ \redcheck \\
$ = \dfrac{11!}{6!5!} \cdot  \dfrac{1!}{1!0!} $ \redcheck \\

=\fbox{$ 462$} \redcheck 
ways \redcheck 

\end{enumerate}

%B

$n_2= 6$ \redcheck \\
$r_2= 4$ \redcheck \\

$ = \dfrac{8!}{8!0!} \cdot \dfrac{6!}{2!4!} \cdot \dfrac{4!}{4!0!} $ \redcheck \\

ways \redcheck

$ = \dfrac{8!}{6!2!} \cdot \dfrac{6!}{5!1!} \cdot \dfrac{4!}{3!1!} $ \redcheck \\
