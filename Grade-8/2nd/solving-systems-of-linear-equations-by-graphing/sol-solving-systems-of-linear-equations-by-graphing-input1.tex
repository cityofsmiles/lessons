\begin{center}
\textbf{Solving Systems of Linear Equations by Graphing}
\end{center} 

\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}
%1
\item $y = \displaystyle  \frac{2}{3}x+6$ \redcheck \\
$-\dfrac{2}{3}x+y =6$ \redcheck \\
$-\dfrac{2}{(3)(6)}x+ \dfrac{y}{6} =\dfrac{6}{6} $ \redcheck \\
\fbox{$\dfrac{x}{-9}+ \dfrac{y}{6} =1 $} \redcheck \\
\fbox{$a_1 = -9$} \redcheck \\
\fbox{$b_1 = 6$} \redcheck \\

$y = -\dfrac{3}{2}x+6$ \redcheck \\
$\dfrac{3}{(2)(6)}x+ \dfrac{y}{6}  =\dfrac{6}{6} $ \redcheck \\
$\dfrac{x}{4}+ \dfrac{y}{6}  =1 $ \redcheck \\
\fbox{$a_2 =4 $} \redcheck \\
\fbox{$b_2 = 6$} \redcheck \\

\vspce 
%2
\item $\dfrac{x}{7}+ \dfrac{y}{7} = \dfrac{7}{7}$ \redcheck \\
$\dfrac{x}{7}+ \dfrac{y}{7} = 1$ \redcheck \\

\fbox{$a_1 =7 $} \redcheck \\
\fbox{$b_1 = 7$} \redcheck \\

$x-y = 1 $ \redcheck \\
$x + \dfrac{y}{-1}  = 1 $ \redcheck \\
\fbox{$a_2 = 1$} \redcheck \\
\fbox{$b_2 = -1$} \redcheck \\

\vspce 
%3
\item $\dfrac{4x}{8} -\dfrac{y}{8}  = \dfrac{8}{8}  $ \redcheck \\
$\dfrac{x}{2} +\dfrac{y}{-8}  = 1 $ \redcheck \\
\fbox{$a_1 = 2$} \redcheck \\
\fbox{$b_1 = -8$} \redcheck \\

$\dfrac{3x}{6} +\dfrac{2y}{6} = \dfrac{6}{6} $ \redcheck \\
$\dfrac{x}{2} +\dfrac{y}{3} = 1$ \redcheck \\
\fbox{$a_2 =2 $} \redcheck \\
\fbox{$b_2 =3 $} \redcheck \\

\vspce 
%4
\item $\dfrac{x}{8} +\dfrac{4y}{8} = \dfrac{8}{8}$ \redcheck \\
$\dfrac{x}{8} +\dfrac{y}{2} = 1$ \redcheck \\
\fbox{$a_1 = 8$} \redcheck \\
\fbox{$b_1 =2 $} \redcheck \\

$ \dfrac{x}{2} -\dfrac{2y}{2} = \dfrac{2}{2} $ \redcheck \\
$ \dfrac{x}{2} +\dfrac{y}{-1} = 1 $ \redcheck \\
\fbox{$a_2 = 2$} \redcheck \\
\fbox{$b_2 = -1$} \redcheck \\

\vspce 
\item $\dfrac{x}{5} +\dfrac{y}{5}  =  \dfrac{5}{5} $ \redcheck \\
$\dfrac{x}{5} +\dfrac{y}{5}  =  1 $ \redcheck \\
\fbox{$a_1 = 5$} \redcheck \\
\fbox{$b_1 = 5$} \redcheck \\

$ y  = 5x+ \dfrac{1}{2}$ \redcheck \\
$2\left[-5x+y  =  \dfrac{1}{2}\right] $ \redcheck \\
$-10x+2y  =  1 $ \redcheck \\
$\dfrac{x}{-\frac{1}{10} }+\dfrac{y}{\frac{1}{2} }  =  1 $ \redcheck \\
\fbox{$a_2 = -\dfrac{1}{10}$} \redcheck \\
\fbox{$b_2 = \dfrac{1}{2} $} \redcheck \\

\end{multicols} 
\end{enumerate}
