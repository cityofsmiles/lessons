\item[4. ] $\dfrac{2y}{2} = \dfrac{6x}{2}   - \dfrac{5}{2} $\redcheck \\
$y = 3x - \dfrac{5}{2} $\redcheck \\
\fbox{$m_1 = 3$}\redcheck \\
\fbox{$b_1 = - \dfrac{5}{2}$}\redcheck \\
$\dfrac{3y}{3} = \dfrac{9x}{3} + \dfrac{1}{3} $\redcheck \\
$y = 3x + \dfrac{1}{3} $\redcheck \\
\fbox{$m_2 = 3$}\redcheck \\
\fbox{$b_2 = \dfrac{1}{3}$}\redcheck \\
$m_1 = m_2$\redcheck \\
$b_1 \neq b_2$\redcheck \\
$\therefore$ \fbox{\small Inconsistent, Independent} \redcheck
\vspce 

\item[5. ] $3x + 5y=15 $ \redcheck \\
$\dfrac{5y}{5} =\dfrac{-3x}{5} + \dfrac{15}{5}  $ \redcheck \\
$y =-\dfrac{3}{5}x + 3$ \redcheck \\
\fbox{$m_1 = -\dfrac{3}{5}$}\redcheck \\
\fbox{$b_1 = 3$}\redcheck \\

$4x - 7y = 10$\redcheck \\
$\dfrac{- 7y}{-7}  =\dfrac{-4x}{-7} +\dfrac{10}{-7} $\redcheck \\
$y=\dfrac{4}{7}x -\dfrac{10}{7} $\redcheck \\
\fbox{$m_2 = \dfrac{4}{7}$}\redcheck \\
\fbox{$b_2 = -\dfrac{10}{7}$}\redcheck \\
$m_1 \neq m_2$\redcheck \\
$b_1 \neq b_2$\redcheck \\
$\therefore$ \fbox{\small Consistent, Independent} \redcheck
