\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}

%1
\item $8x + 2y =7$ \redcheck \\
$\dfrac{2y}{2}  =\dfrac{-8x}{2} + \dfrac{7}{2} $ \redcheck \\
$y=-4x +\dfrac{7}{2} $ \redcheck \\
\fbox{$m_1 = -4$}\redcheck \\
\fbox{$b_1 = \dfrac{7}{2}$}\redcheck \\
$y=-4x+1$ \redcheck \\
\fbox{$m_2 = -4$}\redcheck \\
\fbox{$b_2 = 1$}\redcheck \\
$m_1 = m_2$\redcheck \\
$b_1 \neq b_2$\redcheck \\
$\therefore$ \fbox{\small Inconsistent, Independent} \redcheck 

\vspce 
%2
\item $x - 2y  =  9$ \redcheck \\
$ \dfrac{- 2y}{-2}   = \dfrac{-x}{-2}+ \dfrac{9}{-2} $ \redcheck \\ 
$y= \dfrac{1}{2}x- \dfrac{9}{2} $ \redcheck \\ 
\fbox{$m_1 = \dfrac{1}{2}$}\redcheck \\
\fbox{$b_1 = - \dfrac{9}{2}$}\redcheck \\
$x + 3y = 14$ \redcheck \\
$\dfrac{3y}{3}  = -\dfrac{x}{3} + \dfrac{14}{3}  $ \redcheck \\
$y  = -\dfrac{1}{3}x + \dfrac{14}{3}  $ \redcheck \\
\fbox{$m_2 =-\dfrac{1}{3} $}\redcheck \\
\fbox{$b_2 = \dfrac{14}{3} $}\redcheck \\
$m_1 \neq m_2$\redcheck \\
$b_1 \neq b_2$\redcheck \\
$\therefore$ \fbox{\small Consistent, Independent} \redcheck 

\vspce 
\item $x + 3y = 8$\redcheck \\
$\dfrac{3y}{3}   = -\dfrac{x}{3} +\dfrac{8}{3} $\redcheck \\
$y = -\dfrac{1}{3}x +\dfrac{8}{3} $\redcheck \\
\fbox{$m_1 = -\dfrac{1}{3} $}\redcheck \\
\fbox{$b_1 = \dfrac{8}{3} $}\redcheck \\
$x - 3y = 8$\redcheck \\
$\dfrac{-3y}{-3}   = \dfrac{-x}{-3} +\dfrac{8}{-3} $\redcheck \\
$y = \dfrac{1}{3}x -\dfrac{8}{3} $\redcheck \\
\fbox{$m_2 = \dfrac{1}{3}$}\redcheck \\
\fbox{$b_2 = -\dfrac{8}{3}$}\redcheck \\
$m_1 \neq m_2$\redcheck \\
$b_1 \neq b_2$\redcheck \\
$\therefore$ \fbox{\small Consistent, Independent} \redcheck
\end{multicols} 
\end{enumerate}
