\item $ 2x-3y = 12$ \\
$ 2x-3y -12=0 $ \redcheck \\
$ 4x+3y = 24 $ \\
$ 4x+3y -24= 0 $ \redcheck \\

\fbox{\parbox{11.5em}{
$\begin{array}{llcl}
 & 2x-3y -12& = & 0  \\
+ & 4x+3y -24& = & 0  \\
\cline{1-4}
 & 6x -36& = & 0 \redcheck \\
\end{array}$\\}}

$ 6x -36 = 0$ \\
$\dfrac{6x}{6} = \dfrac{36}{6} $ \redcheck \\
\fbox{$ x = 6 $ }\redcheck \\

$ 2x-3y = 12 $ \redcheck \\
$ 2(6)-3y = 12 $ \redcheck \\
$ 12-3y = 12 $ \redcheck \\
$ -3y = 12 -12$ \redcheck \\
$ -3y = 0$ \redcheck \\
\fbox{$ y=0 $ }\redcheck \\

$\therefore$ Solution = \fbox{$\left( 6,0 \right)$ } \redcheck

\item $ 5x+11 = 7y $  \\
$ 3[5x-7y+11 =  0]$ \redcheck \\
\fbox{$ 15x-21y+33 =  0$ }\redcheck \\

$ 8y-18 = 3x $ \\
$ 5[-3x+8y-18 = 0 ]$ \redcheck \\
\fbox{$ -15x+40y-90 = 0 $ }\redcheck \\

\fbox{\parbox{12.5em}{
$\begin{array}{lrcl}
 & 15x-21y+33 & = & 0  \\
+ & -15x+40y-90 & = & 0  \\
\cline{1-4}
 & 19y -57& = & 0 \redcheck \\
\end{array}$\\}}

$ 19y -57=0$  \\
$ \dfrac{19y}{19}  =\dfrac{57}{19} $ \redcheck \\
\fbox{$ y=3$ }\redcheck \\

$ 5x+11 = 7y$ \redcheck \\
$ 5x+11 = 7(3) $ \redcheck \\
$ 5x+11 = 21 $ \redcheck \\
$ 5x = 21-11 $ \redcheck \\
$ \dfrac{5x}{5}  = \dfrac{10}{5} $ \redcheck \\
\fbox{$ x=2 $ }\redcheck \\

$\therefore$ Solution = \fbox{$\left( 2,3 \right)$ } \redcheck
