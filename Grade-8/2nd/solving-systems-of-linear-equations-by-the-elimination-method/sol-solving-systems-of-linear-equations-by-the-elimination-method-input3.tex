\item[4. ] $x  =  y+1 $ \\
\fbox{$ x-y-1=0$ }\redcheck \\

$ x-y  = 1$ \\
$ -1[x-y-1=0]$ \redcheck \\
\fbox{$ -x+y +1 = 0$ }\redcheck \\

\fbox{\parbox{11.5em}{
$\begin{array}{lrcl}
 & x-y -1& = & 0  \\
+ & -x+y +1& = & 0  \\
\cline{1-4}
 & 0& = & 0 \redcheck \\
\end{array}$\\}}

$\therefore$ Solution = \fbox{None} \redcheck

\item[5. ] $ 2x+y = 4 $ \\
\fbox{$ 2x+y-4 = 0$ }\redcheck \\

$ x+2y = 4 $ \\
$ -2[x+2y-4 = 0] $ \redcheck \\
\fbox{$ -2x-4y+8=0$ }\redcheck \\

\fbox{\parbox{11.5em}{
$\begin{array}{lrcl}
 & 2x+y-4& = & 0  \\
+ & -2x-4y+8& = & 0  \\
\cline{1-4}
 & -3y +4& = & 0 \redcheck \\
\end{array}$\\}}

$ -3y +4=0 $ \\
$ \dfrac{-3y}{-3}  =\dfrac{-4}{-3}  $ \redcheck \\
\fbox{$ y= \dfrac{4}{3} $ }\redcheck \\

$ 2x+y = 4 $ \redcheck \\
$ 2x+\dfrac{4}{3} = 4 $ \redcheck \\
$ 2x = 4 - \dfrac{4}{3} $ \redcheck \\
$ \dfrac{1}{2}\left[2x = \dfrac{8}{3}\right] $ \redcheck \\
\fbox{$ x=\dfrac{4}{3}$ }\redcheck \\

$\therefore$ Solution = \fbox{$\left( \dfrac{4}{3}, \dfrac{4}{3}\right)$ } \redcheck
