\textbf{Relation:} a set of ordered pairs

\vspce 

\textbf{Domain:} the first set of all elements in an ordered pair

\vspce 

\textbf{Range:} the second  set of all elements in an ordered pair

\vspce 

\textbf{Representations of Relations:} 
\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}
\item \hspce Ordered Pairs
\item \hspce Table
\item \hspce Graph 
\item \hspce Rule
\item \hspce Mapping Diagram 
\end{multicols} 
\end{enumerate}  

\vspce 

\textbf{Kinds of Relation:} 
\begin{enumerate}[label = \arabic*. ]
\item \hspce One-to-one Correspondence: Each element in the first set is paired with exactly one element in the second set. 
\item \hspce Many-to-one Correspondence: Many elements in the first set are paired with the same elements in the second set. 
\item \hspce One-to-many Correspondence: One element of the first set is paired with different  elements in the second set. 
\end{enumerate}  
}
