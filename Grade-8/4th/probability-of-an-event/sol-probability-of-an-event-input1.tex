\textbf{Probability of an Event}\\

\vspace*{1ex}

\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}
%1
\item %Three coins are tossed. What is the probability of getting:
%
\begin{enumerate}[label = \alph*. ]
%a
\item P(two heads) \redcheck \\
$=\dfrac{3}{8} $ \redcheck 

%b
\item P(at least two heads) \redcheck \\
$=\dfrac{4}{8} $ \redcheck \\
$=\dfrac{1}{2} $ \redcheck 

%c
\item P(no tail) \redcheck \\
$=\dfrac{1}{8} $ \redcheck 

\end{enumerate} 
%\end{multicols} 
%2
\item %A pair of dice is rolled. What is the probability of getting: 
\begin{enumerate}[label = \alph*. ]
%a
\item P(sum of five) \redcheck \\
$=\dfrac{4}{36} $ \redcheck \\
$=\dfrac{1}{9} $ \redcheck 

%b
\item P(sum is prime) \redcheck \\
$=\dfrac{15}{36} $ \redcheck \\
$=\dfrac{5}{12} $ \redcheck 

%c
\item P(sum greater  than 9)\redcheck \\
$=\dfrac{6}{36} $ \redcheck \\
$=\dfrac{1}{6} $ \redcheck 

%d
\item P(not a double) \redcheck \\
$=\dfrac{30}{36} $ \redcheck \\
$=\dfrac{5}{6} $ \redcheck 

\end{enumerate} 
%3
\item %Given a regular deck of cards, a card is drawn at random. What is the probability of getting:
\begin{enumerate}[label = \alph*. ]
%a
\item P(a red number card) \redcheck \\
$=\dfrac{18}{52} $ \redcheck \\
$=\dfrac{9}{26} $ \redcheck 

%b
\item P(not a heart) \redcheck \\
$=\dfrac{39}{52} $ \redcheck \\
$=\dfrac{3}{4} $ \redcheck 

%c
\item P(a black ace) \redcheck \\
$=\dfrac{2}{52} $ \redcheck \\
$=\dfrac{1}{26} $ \redcheck 

\end{enumerate} 
%4
\item %A letter is drawn at random from those in MATHEMATICS. Find the probability of each event.
\begin{enumerate}[label = \alph*. ]
%a
\item P(letter M) \redcheck \\
$=\dfrac{2}{11} $ \redcheck 

%b
\item P(comes from the first half of the alphabet) \redcheck \\
$=\dfrac{8}{11} $ \redcheck 

%c
\item P(letter C) \redcheck \\
$=\dfrac{1}{11} $ \redcheck 
\end{enumerate} 
\end{multicols}
\end{enumerate}
