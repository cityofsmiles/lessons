%\vspce

%Find the quotient of the following rational algebraic expressions. 

\vspce

\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}

%#1
\item %\hspce $\displaystyle \frac{14x^2}{20y^2} \div \frac{56x^2}{y}$ 
$\displaystyle \frac{14x^2}{20y^2} \cdot  \frac{y}{56x^2}$ \\
$=\dfrac{1}{80 y}$
%\vspce
%#2
\item %\hspce $\displaystyle \frac{4a-4b}{30a^2} \div \frac{a-b}{9a}$ 
%\vspce
$\displaystyle \frac{4a-4b}{30a^2} \cdot  \frac{9a}{a-b}$ \\
$\displaystyle \frac{4(a-b) }{30a^2} \cdot  \frac{9a}{a-b}$ \\
$=\dfrac{6}{5 a}$
\end{multicols}
%#3
\item %\hspce $\displaystyle \frac{x^2-4}{x^2+4x+4} \div \frac{x^2-x-2}{x^2+3x+2}$
%\vspce
$\displaystyle \frac{x^2-4}{x^2+4x+4} \cdot \frac{x^2+3x+2}{x^2-x-2}$\\
$\displaystyle \frac{\left(x - 2\right) \left(x + 2\right)}{\left(x + 2\right)\left(x + 2\right)} \cdot \frac{\left(x + 1\right) \left(x + 2\right)}{\left(x - 2\right) \left(x + 1\right)}$\\
$=1$
%#4
\item %\hspce $\displaystyle \frac{x^2-2x-3}{x^{2}-3x} \div \frac{x^2-4}{x^{2}+2x}$
%\vspce
$\displaystyle \frac{x^2-2x-3}{x^{2}-3x} \cdot \frac{x^{2}+2x}{x^2-4}$\\
$\displaystyle \frac{\left(x - 3\right) \left(x + 1\right)}{x \left(x - 3\right)} \cdot \frac{x \left(x + 2\right)}{\left(x - 2\right) \left(x + 2\right)}$\\
$=\dfrac{x + 1}{x - 2}$
%#5
\item %\hspce $\displaystyle \frac{x^2-4}{x^2+2x} \div \frac{x^2+x-6}{2x+4}$
$\displaystyle \frac{x^2-4}{x^2+2x} \cdot \frac{2x+4}{x^2+x-6}$\\
$\displaystyle \frac{\left(x - 2\right) \left(x + 2\right)}{x \left(x + 2\right)} \cdot \frac{2 \left(x + 2\right)}{\left(x - 2\right) \left(x + 3\right)}$\\
$=\dfrac{2 \left(x + 2\right)}{x \left(x + 3\right)}$
%%
\end{enumerate}
