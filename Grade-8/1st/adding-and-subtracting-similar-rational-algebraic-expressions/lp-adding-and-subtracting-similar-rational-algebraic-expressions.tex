
\def \LessonDay {Adding and Subtracting Similar Rational Algebraic Expressions}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Describe the steps in adding and subtracting similar rational algebraic expressions; 
\item Perform the steps in adding and subtracting similar rational algebraic expressions; and, 
\item Project independence and willingness in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems by adding and subtracting similar rational algebraic expressions.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of adding and subtracting similar rational algebraic expressions? 
%2
\item How can you apply your knowledge of adding and subtracting similar rational algebraic expressions in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we add or subtract similar rational algebraic expressions? 
%2
\item How do we solve problems involving algebraic expressions that require adding and subtracting similar rational algebraic expressions? 
\end{enumerate}   
}
