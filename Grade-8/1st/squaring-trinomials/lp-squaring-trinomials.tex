\def \ExamplesDay {Some examples of squaring-trinomials are the following: 
\begin{enumerate}[label = \arabic*. ]
\item \hspce $(x - y)^{2} = x^{2} - 2xy + y^2$

\item \hspce $(m + 2n^2)^{2} = m^2 + 4mn^2 + 4n^4 $
%3
\item \hspce $(-2a^3 - 3b^2)^{2} = 4a^6 + 12a^3b^2 + 9b^4 $
\end{enumerate}
