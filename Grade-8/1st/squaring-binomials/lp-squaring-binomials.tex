
\def \LessonDay {\makecell{Squaring Binomials}}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Demonstrate the steps in squaring binomials; 
\item Determine the missing  terms of a perfect square trinomial; and, 
\item Display determination and independence in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving squaring binomials.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of squaring binomials? 
%2
\item How can you apply your knowledge of squaring binomials in solving these real life problems? 
%3
%\item 
\end{enumerate}   
}

\begin{enumerate}[label = \arabic*. ]

%3
%\item

}

\def \ExamplesDay {Some examples of squaring-binomials are the following:

\item \hspce $(x - y)^{2} = x^{2} - 2xy + y^2$

\item \hspce $(m + 2n^2)^{2} = m^2 + 4mn^2 + 4n^4 $

\end{enumerate}
