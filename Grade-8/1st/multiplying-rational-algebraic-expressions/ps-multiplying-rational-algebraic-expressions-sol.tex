%Find the product of the following rational algebraic expressions. 

%\vspce

\begin{enumerate}[label = \arabic*. ]
%begin{multicols}{2}

%#1
\item% \hspce $\displaystyle \frac{12mn^2}{6xy^2} \cdot \frac{9x^2y^2}{4m^2n^2}$ \\
%\vspce
$\dfrac{9 x}{2 m}$
%#2
\item %\hspce $\displaystyle \frac{x^2-y^2}{2xy} \cdot \frac{y^2}{x+y}$ 
%\vspce
$\dfrac{\left(x - y\right) \left(x + y\right)}{2xy} \cdot \dfrac{y^2}{x+y} $\\
$=\dfrac{y \left(x - y\right)}{2 x} $

%#3
\item% \hspce $\displaystyle \frac{x^2-2x}{x^{2}-3x-10} \cdot \frac{x^2-4x-5}{x^2-4}$
%\vspce
$\dfrac{x(x-2)}{\left(x - 5\right) \left(x + 2\right)} \cdot \dfrac{\left(x - 5\right) \left(x + 1\right)}{\left(x - 2\right) \left(x + 2\right)} $\\
$=\dfrac{x \left(x + 1\right)}{\left(x + 2\right)^{2}} $

%#4
\item %\hspce $\displaystyle \frac{x^2+4x+4}{x^{2}+3x+2} \cdot \frac{x^2-1}{x^{2}-4}$
%\vspce
$\dfrac{\left(x + 2\right)\left(x + 2\right)}{\left(x + 1\right) \left(x + 2\right)} \cdot \dfrac{\left(x - 1\right) \left(x + 1\right)}{\left(x - 2\right) \left(x + 2\right)} $\\
$=\dfrac{x - 1}{x - 2} $

%#5
\item %\hspce $\displaystyle \frac{a^2+2ab+b^2}{a^2-b^2} \cdot \frac{a-b}{a+b}$
$\dfrac{\left(a + b\right)\left(a + b\right)}{\left(a - b\right) \left(a + b\right)} \cdot \dfrac{a-b}{a+b} $\\
$=1 $

%end{multicols}
\end{enumerate}
