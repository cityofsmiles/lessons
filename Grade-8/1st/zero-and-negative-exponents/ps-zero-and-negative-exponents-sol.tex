\textbf{Problem Set}

\vspce

%Convert the following expressions into expressions with positive exponents.
  
\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}

\item \hspce %$a^{-7}$
$\displaystyle \frac{1}{a^{7}}$

\item \hspce% $\displaystyle \frac{a^{-4}}{e^{-5}}$
$\displaystyle \frac{\displaystyle \frac{1}{a^{4}} }{\displaystyle \frac{1}{e^{5}} }$\\
$=\displaystyle \frac{1}{a^{4}} \div \displaystyle \frac{1}{e^{5}}$\\
$=\left( \displaystyle \frac{1}{a^{4}}\right)  (e^{5}) $\\
$=\displaystyle \frac{e^{5}}{a^{4}}$

\item \hspce %$x^{-2}y^{-5}$
$\left( \displaystyle \frac{1}{x^{2}} \right)\left( \displaystyle \frac{1}{y^{5}} \right)  $%\\
$=\displaystyle \frac{1}{x^{2} y^{5}}$

\item \hspce %$x^{2}y^{-5}z$
$(x^{2}) \left( \displaystyle \frac{1}{y^{5}}\right) (z) $%\\
$=\displaystyle \frac{x^{2} z}{y^{5}} $

\item \hspce %$mn^{-6}p$
$(m) \left( \displaystyle \frac{1}{n^{6}}\right) (p) $%\\
$=\displaystyle \frac{m p}{n^{6}}$

\item \hspce% $\displaystyle \frac{(a-b)}{(a+b)^{0}}$
$\displaystyle \frac{(a-b)}{1}$%\\
$ =a - b$

\item \hspce %$x^{-7}y^{2}z^{-4}$
$\left( \displaystyle \frac{1}{x^{7}} \right)(y^{2}) \left( \displaystyle \frac{1}{z^{4}} \right) $\\
$=\displaystyle \frac{y^{2}}{x^{7} z^{4}}$


\item \hspce %$a^{-4}b^{5}c^{-2}$
$\left( \displaystyle \frac{1}{a^{4}} \right) (b^{5}) \left( \displaystyle \frac{1}{c^{2}} \right) $\\
$=\displaystyle \frac{b^{5}}{a^{4} c^{2}}$

\item \hspce %$m^{-3}n^2p^{-8}$
$\left( \displaystyle \frac{1}{m^{3}} \right) (n^2)\left( \displaystyle \frac{1}{p^{8}} \right) $\\
$=\displaystyle \frac{n^{2}}{m^{3} p^{8}}$

\item \hspce %$\displaystyle \frac{(m^2-n)}{(m^2+n)^{0}}$
$\displaystyle \frac{(m^2-n)}{1}$
$=\displaystyle m^{2} - n$

\end{multicols}
\end{enumerate}
