
\def \LessonDay {Factoring the Sum or Difference of Two Cubes}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Identify the cube roots of algebraic expressions; 
\item Find the factors of the sum or difference of two cubes; and, 
\item Show independence and independence in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems by factoring the sum or difference of two cubes.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of factoring  the sum or difference of two cubes? 
%2
\item How can you apply your knowledge of factoring the sum or difference of two cubes in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we find the  factors of the sum or difference of two cubes? 
%2
\item How do we solve problems involving algebraic expressions that require  factoring the sum or difference of two cubes? 
\end{enumerate}   
}
