%Factor the following polynomials completely.  
\begin{enumerate} 
\begin{multicols}{2}
\item \hspce %$3x^{2} + 7x + 4 $
$3x^{2}+3x +4x+4 $\\
$=(3x^{2}+3x) +(4x+4) $\\
$=3x(x+1) +4(x+1) $\\
$= \left(x + 1\right) \left(3 x + 4\right)$

\item \hspce %$2x^{2} + 3x - 9$
$2x^{2} -3x+6x - 9$\\
$=(2x^{2} -3x) +(6x - 9) $\\
$=x(2x -3) +3(2x -3)$\\
$=\left(2 x - 3\right)\left(x + 3\right)$

% 
\item \hspce %$6a^{2} + 11a + 3		$
$6a^{2} + 2a+9a + 3$\\
$=(6a^{2} + 2a) +(9a + 3) $\\
$=2a(3a + 1) +3(3a + 1)$\\
$=  \left(3 a + 1\right)\left(2 a + 3\right)$

% 
\item \hspce %$4m^{2} - 5m - 6$
% 
$4m^{2} - 8m+3m - 6$\\
$=(4m^{2} - 8m) +(3m - 6) $\\
$=4m(m - 2) +3(m - 2)$\\
$= \left(m - 2\right) \left(4 m + 3\right)$

\item \hspce %$3a^{2} - a - 4$
% 
$3a^{2} - 4a+3a - 4$\\
$=(3a^{2} - 4a) +(3a - 4) $\\
$=a(3a - 4) +(3a - 4)$\\
$= \left(3 a - 4\right)\left(a + 1\right) $

\item \hspce% $4 m^{2} - 11 m n + 6 n^{2} $
$4 m^{2} - 8 m n- 3 m n + 6 n^{2}$\\
$=(4 m^{2} - 8 m n) -( 3 m n - 6 n^{2}) $\\
$=4 m(m- 2n) -3n( m- 2n)$\\
$=(m-2n) (4m-3n)  $


% 

\item \hspce %$ 3 a^{2} - 7 a b - 6 b^{2}$
$ 3 a^{2} - 9 a b+2a b - 6 b^{2}$\\
$=(3 a^{2} - 9 a b) +(2a b - 6 b^{2}) $\\
$=3 a(a - 3 b) +2b(a - 3 b)$\\
$= (a-3b)(3a+2b)  $

% 

\item \hspce %$ 4 c^{2} - 19 c d - 5 d^{2}$
$ 4 c^{2} - 20 c d+cd - 5 d^{2}$\\
$=(4 c^{2} - 20 c d) +(cd - 5 d^{2}) $\\
$=4c(c - 5d) +d(c - 5 d)$\\
$= (c-5d) (4c+d) $

% 

\item \hspce %$10 x^{2} - 27 x y + 18 y^{2} $
$10 x^{2} - 15 x y- 12 x y + 18 y^{2}  $\\
$=(10 x^{2} - 15 x y) -( 12 x y - 18 y^{2}) $\\
$=5x(2 x - 3 y) -6y(2 x - 3 y)$\\
$=(2x-3y) (5x-6y)  $

% 

\item \hspce% $6 m^{2} - 7 m n - 3 n^{2} $
$ 6 m^{2} - 9 m n +2mn- 3 n^{2}$\\
$=(6 m^{2} - 9 m n) +(2mn- 3 n^{2}) $\\
$=3m(2m - 3 n) +n(2m-3n)$\\
$= (2m-3n)(3m+n)  $

% 
\end{multicols} 
 
\end{enumerate}
