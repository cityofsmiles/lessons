
\def \LessonDay {Factoring General Trinomials}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Define the steps in factoring general trinomials; 
\item Calculate the factors of  general trinomials; and, 
\item Display perseverance and perseverance in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems by factoring general trinomials.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of factoring general trinomials? 
%2
\item How can you apply your knowledge of factoring general trinomials in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we  factor general trinomials? 
%2
\item How do we solve problems involving algebraic expressions that require  factoring general trinomials? 
\end{enumerate}   
}
