\textbf{Problem Set}

\vspce

Factor the following polynomials completely.  
\begin{enumerate} 


\item \hspce $3x^{2} + 7x + 4 $
% \left(x + 1\right) \left(3 x + 4\right)
\item \hspce $2x^{2} + 3x - 9$
% \left(x + 3\right) \left(2 x - 3\right)
\item \hspce $6a^{2} + 11a + 3		$
% \left(2 a + 3\right) \left(3 a + 1\right)
\item \hspce $4m^{2} - 5m - 6$
% \left(m - 2\right) \left(4 m + 3\right)
\item \hspce $3a^{2} - a - 4$
% \left(a + 1\right) \left(3 a - 4\right)
\item \hspce $4 m^{2} - 11 m n + 6 n^{2} $
% (m-2n) (4m-3n) 

\item \hspce $ 3 a^{2} - 7 a b - 6 b^{2}$
% (3a+2b) (a-3b) 

\item \hspce $ 4 c^{2} - 19 c d - 5 d^{2}$
% (4c+d) (c-5d) 

\item \hspce $10 x^{2} - 27 x y + 18 y^{2} $
% (2x-3y) (5x-6y) 

\item \hspce $6 m^{2} - 7 m n - 3 n^{2} $
% (3m+n) (2m-3n) 

 
\end{enumerate}
