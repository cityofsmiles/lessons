
\def \LessonDay {Factoring by Grouping}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Distinguish the steps in factoring by grouping with factoring the greatest common monomial factor; 
\item Solve the factors of an algebraic expression using the grouping method; and, 
\item Demonstrate willingness and willingness in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems by using factoring by grouping.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of factoring by grouping? 
%2
\item How can you apply your knowledge of factoring by grouping in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we  factor  by grouping? 
%2
\item How do we solve problems involving algebraic expressions that require  factoring by grouping? 
\end{enumerate}   
}
