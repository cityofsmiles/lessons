
\def \LessonDay {Factoring Perfect Square Trinomials}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item State the steps in factoring perfect square trinomials; 
\item Calculate the square roots of given  algebraic expressions; and, 
\item Display independence and independence in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems by factoring perfect square trinomials.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of factoring perfect square trinomials? 
%2
\item How can you apply your knowledge of factoring perfect square trinomials in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we  factor perfect square trinomials? 
%2
\item How do we solve problems involving algebraic expressions that require  factoring perfect square trinomials? 
\end{enumerate}   
}
