
\def \LessonDay {Simplifying Rational Expressions}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Discuss the steps in simplifying rational expressions; 
\item Simplify rational algebraic expressions; and, 
\item Exhibit enjoyment and interest in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving simplifying rational expressions.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of simplifying rational expressions? 
%2
\item How can you apply your knowledge of simplifying rational expressions in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we simplify rational expressions? 
%2
\item How do we solve problems involving algebraic expressions that require simplifying rational expressions? 
\end{enumerate}   
}
