\textbf{Problem Set}

\vspce

%Simplify the following  rational expressions.

\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}

\item %\hspce %$\displaystyle \frac{x^{2}+3x}{x+3}$
$\displaystyle \frac{x(x+3)}{x+3} $\\
$= x$

\item %\hspce %$\displaystyle \frac{2x^{2}+8x}{2x}$
$\displaystyle \frac{2x(x+4)}{2x} $\\
$= x + 4$

\item% \hspce %$\displaystyle \frac{x^{2}-16}{x-4}$
$\displaystyle \frac{(x-4) (x+4) }{x-4}$\\
$= x + 4$

\item% \hspce %$\displaystyle \frac{x^{3}+64}{x^{2}-4x+16}$
$\displaystyle \frac{(x+4)(x^{2}-4x+16)}{x^{2}-4x+16}$\\
$= x + 4$

\item %\hspce %$\displaystyle \frac{2x^{2}-9x-5}{x-5}$
$\displaystyle \frac{(2 x + 1) (x-5) }{x-5}$\\
$=2 x + 1 $


\item %\hspce %$\displaystyle \frac{2x^{2}+4x}{x+2}$
$\displaystyle \frac{2 x(x+2) }{x+2}$\\
$= 2 x$

\item %\hspce %$\displaystyle \frac{4x^{3}-8x^2}{4x^2}$
$\displaystyle \frac{4x^2(x - 2) }{4x^2}$\\
$= x - 2$

\item %\hspce %$\displaystyle \frac{4x^{2}-25}{2x-5}$
$\displaystyle \frac{(2x-5) (2x+5) }{2x-5}$\\
$= 2 x + 5$
\end{multicols}

\item %\hspce %$\displaystyle \frac{8x^{3}-27}{4x^{2}+6x+9}$
$\displaystyle \frac{(2 x - 3) (4x^{2}+6x+9) }{4x^{2}+6x+9}$
$= 2 x - 3$

\item %\hspce %$\displaystyle \frac{3x^{2}-12x+12}{x-2}$
$\displaystyle \frac{(3 x - 6) (x-2) }{x-2}$
$= 3 x - 6$


%\end{multicols}
\end{enumerate}
