\begin{enumerate}[label = \arabic*. ]
%begin{multicols}{2}

\item \hspce $\displaystyle \frac{4x^{2}-4x+1}{2x-1}$
\item \hspce $\displaystyle \frac{x^{2}+5x+6}{x+3}$

%end{multicols}
\end{enumerate}
