
\def \LessonDay {Rational Algebraic Expressions}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Define rational algebraic expressions; 
\item Rewrite expressions into rational algebraic expressions with positive exponents; and, 
\item Show enjoyment and willingness in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving rational algebraic expressions.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of rational algebraic expressions? 
%2
\item How can you apply your knowledge of rational algebraic expressions in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, what are rational algebraic expressions? 
%2
\item How do we solve problems involving rational algebraic expressions? 
\end{enumerate}   
}
