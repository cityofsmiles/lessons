\begin{enumerate}[label = \arabic*. ]
\begin{multicols}{2}

%#1
\item %\hspce  $\displaystyle \frac{a}{a-b} - \frac{b}{a+b}$ 
%\vspce
$\dfrac{a(a+b) - b(a-b)}{\left(a - b\right) \left(a + b\right)}$\\
$=\dfrac{a^{2} +ab -ab+ b^{2}}{\left(a - b\right) \left(a + b\right)}$\\
$=\dfrac{a^{2} + b^{2}}{\left(a - b\right) \left(a + b\right)}$

%#2
\item %\hspce  $\displaystyle \frac{3}{2x+1}+\frac{5}{3x-2}$ 
%\vspce
$\dfrac{3(3x-2)+5(2x+1)}{\left(2 x + 1\right) \left(3 x - 2\right)}$\\
$=\dfrac{9x-6+10x+5}{\left(2 x + 1\right) \left(3 x - 2\right)}$\\
$=\dfrac{19 x - 1}{\left(2 x + 1\right) \left(3 x - 2\right)}$
%#3
\item %\hspce  $\displaystyle \frac{3a+12}{2a-8}+\frac{a+4}{a-4}$
%\vspce
$\dfrac{3a+12+2(a+4)}{2 \left(a - 4\right)}$\\
$=\dfrac{3a+12+2a+8}{2 \left(a - 4\right)}$\\
$=\dfrac{5a+20}{2 \left(a - 4\right)}$\\
$=\dfrac{5 \left(a + 4\right)}{2 \left(a - 4\right)}$
%#4
\item %\hspce  $\displaystyle \frac{y+1}{y} + \frac{y-1}{y+1}$
%\vspce
$\dfrac{(y + 1)(y + 1) + y(y - 1)}{y \left(y + 1\right)}$\\
$=\dfrac{y^{2} + 2 y + 1 + y^2 - y}{y \left(y + 1\right)}$\\
$=\dfrac{2 y^{2} + y + 1}{y \left(y + 1\right)}$
%#5
\item %\hspce $\displaystyle \frac{2x}{x^2-4x+4} - \frac{1}{x-2}$
$\dfrac{2x - (x+2)}{\left(x - 2\right)\left(x - 2\right)}$\\
$=\dfrac{2x - x-2}{\left(x - 2\right)\left(x - 2\right)}$\\
$=\dfrac{x + 2}{\left(x - 2\right)^{2}}$
\end{multicols}
\end{enumerate}
