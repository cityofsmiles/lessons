
\def \LessonDay {Evaluating Rational Expressions}

\def \LearningCompetenciesDay {}

\def \ObjectivesDay {
\item Reiterate the steps in evaluating rational expressions; 
\item Evaluate rational algebraic expressions; and, 
\item Show independence and perseverance in solving problems.
}

\def \PurposeDay {The purpose of this lesson is to enable the students to solve real life problems involving evaluating rational expressions.}  

\def \ApplicationDay { Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In what real life situations or problems can we observe some examples of evaluating rational expressions? 
%2
\item How can you apply your knowledge of evaluating rational expressions in solving these real life problems? 
\end{enumerate}   
}

\def \GeneralizationDay {Let the students answer the following questions: 
\begin{enumerate}[label = \arabic*. ]
%1
\item In your own words, how do we evaluate rational expressions? 
%2
\item How do we solve problems involving algebraic expressions that require evaluating rational expressions? 
\end{enumerate}   
}
