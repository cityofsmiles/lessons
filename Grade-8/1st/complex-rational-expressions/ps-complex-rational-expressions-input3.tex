%\vspce

\begin{enumerate}[label = \arabic*. ]

%#1

%\vspce
%#2

%\vspce
%#3
\item \hspce $\displaystyle \frac{x+\displaystyle \frac{x}{y}}{y-\displaystyle \frac{y}{x}}$
%\vspce
%#4
\item \hspce $\displaystyle \frac{\displaystyle\frac{1}{a-2} - \displaystyle\frac{3}{a-1}}{\displaystyle\frac{5}{a-2} + \displaystyle\frac{2}{a-1}}$
%\vspce
%#5
\item \hspce $\displaystyle \frac{\displaystyle\frac{y+1}{y}}{\displaystyle\frac{y-1}{2y}}$ 
\end{multicols}
\end{enumerate}
